00925 // This function fully formats the zcl buffer given all the necessary data.
00926 // The format string describes how the subsequent arguments are to be
00927 // written into the zcl buffer.  The format characters are the same as
00928 // those used by the command interpreter, namely:
00929 // 'u'   A one-byte unsigned value.
00930 // 'v'   A two-byte unsigned value encoded low byte first.
00931 // 'w'   A four-byte unsigned value encoded from low to high byte.
00932 // 'b'   A sequence of unsigned bytes.  The first supplied value is a
00933 //       pointer to the data and the second value is the number of bytes.
00934 
00935 void makeZclBuffer(int8u frameControl,
00936                    int16u clusterId,
00937                    int8u commandId,
00938                    char *format,
00939                    ...)
00940 {
00941   int8u i = 0;
00942   va_list argPointer;
00943   zclBufferSetup(frameControl, clusterId, commandId);
00944   va_start(argPointer, format);
00945   
00946   while (format[i] != 0) {
00947     char command = format[i];
00948     switch (command) {
00949     case 'u':
00950       zclBufferAddByte(va_arg(argPointer, int));
00951       break;
00952     case 'v':
00953       zclBufferAddWord(va_arg(argPointer, int));
00954       break;
00955     case 'w':
00956       zclBufferAddInt32(va_arg(argPointer, int));
00957       break;
00958     case 'b': {
00959       int8u *contents = va_arg(argPointer, int8u *);
00960       int8u count = va_arg(argPointer, int);
00961       zclBufferAddByte(count);
00962       MEMCOPY(appZclBuffer + appZclBufferLen, contents, count);
00963       appZclBufferLen += count;
00964       break;
00965     }
00966     default:
00967       assert(FALSE);
00968     }
00969     i += 1;
00970   }
00971 
00972   va_end(argPointer);
00973   cliBufferPrint();
00974 }
