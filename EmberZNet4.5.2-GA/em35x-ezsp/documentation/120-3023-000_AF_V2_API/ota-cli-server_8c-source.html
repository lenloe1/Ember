00142 // zcl ota server notify <destination> <dst endpoint> <payload type>
00143 //   <queryJitter> <manufacturer-id> <image-type-id> <firmware-version>
00144 // Payload Type:
00145 //   0 = jitter value only
00146 //   1 = jitter and manufacuter id
00147 //   2 = jitter, manufacuter id, and device id
00148 //   3 = jitter, manufacuter id, device id, and firmware version
00149 //
00150 // The CLI requires all parameters, but if the payload type does not
00151 // require that parameter, simply set it to 0.
00152 void otaImageNotifyCommand(void)
00153 {
00154   EmberAfOtaImageId id;
00155   EmberNodeId dest = emberUnsignedCommandArgument(0);
00156   int8u endpoint = emberUnsignedCommandArgument(1);
00157   int8u payloadType = emberUnsignedCommandArgument(2);
00158   int8u jitter = emberUnsignedCommandArgument(3);
00159   id.manufacturerId = emberUnsignedCommandArgument(4);
00160   id.imageTypeId = emberUnsignedCommandArgument(5);
00161   id.firmwareVersion = emberUnsignedCommandArgument(6);
00162 
00163   emberAfOtaServerSendImageNotifyCallback(dest,
00164                                           endpoint,
00165                                           payloadType,
00166                                           jitter,
00167                                           &id);
00168 }

00172 static void setPolicy(void)
00173 {
00174   int8u value = emberUnsignedCommandArgument(0);
00175   if (emberCurrentCommand->name[0] == 'q') {
00176     emAfOtaServerSetQueryPolicy(value);
00177   } else if (emberCurrentCommand->name[0] == 'b') {
00178     emAfOtaServerSetBlockRequestPolicy(value);
00179   } else if (emberCurrentCommand->name[0] == 'u') {
00180     emAfOtaServerSetUpgradePolicy(value);
00181   } else if (emberCurrentCommand->name[0] == 'p') {
00182     if (emberCurrentCommand->name[9] == 'm') {
00183       emAfSetPageRequestMissedBlockModulus(value);
00184     } else if (emberCurrentCommand->name[9] == 's') {
00185       emAfOtaServerSetPageRequestPolicy(value);
00186     }
00187   }
00188 }
