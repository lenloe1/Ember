00044 void emAfOtaPrintAllImages(void)
00045 {
00046   int8u i = 0;
00047   EmberAfOtaImageId id = emberAfOtaStorageIteratorFirstCallback();
00048   while (emberAfIsOtaImageIdValid(&id)) {
00049     EmberAfOtaHeader header;
00050     EmberAfTagData tagInfo[MAX_TAGS];
00051     int16u totalTags;
00052     otaPrintln("Image %d", i);
00053     if (EMBER_AF_OTA_STORAGE_SUCCESS
00054         != emberAfOtaStorageGetFullHeaderCallback(&id,
00055                                                   &header)) {
00056       otaPrintln("  ERROR: Could not get full header!");
00057     } else {
00058       otaPrintln("  Header Version: 0x%2X", header.headerVersion);
00059       otaPrintln("  Header Length:  %d bytes", header.headerLength);
00060       otaPrintln("  Field Control:  0x%2X", header.fieldControl);
00061       emberAfOtaBootloadClusterFlush();
00062       otaPrintln("  Manuf ID:       0x%2X", header.manufacturerId);
00063       otaPrintln("  Image Type:     0x%2X", header.imageTypeId);
00064       otaPrintln("  Version:        0x%4X", header.firmwareVersion);
00065       emberAfOtaBootloadClusterFlush();
00066       otaPrintln("  Zigbee Version: 0x%2X", header.zigbeeStackVersion);
00067       otaPrintln("  Header String:  %s",    header.headerString);
00068       otaPrintln("  Image Size:     %l bytes",    header.imageSize);
00069       emberAfOtaBootloadClusterFlush();
00070       if (headerHasSecurityCredentials(&header)) {
00071         otaPrintln("  Security Cred:  0x%X", header.securityCredentials);
00072       }
00073       if (headerHasUpgradeFileDest(&header)) {
00074         emberAfOtaBootloadClusterPrint("  Upgrade Dest:   ");
00075         emberAfPrintBigEndianEui64(header.upgradeFileDestination);
00076         emberAfOtaBootloadClusterFlush();
00077         otaPrintln("");
00078       }
00079       if (headerHasHardwareVersions(&header)) {
00080         otaPrintln("  Min. HW Ver:    0x%2X", header.minimumHardwareVersion);
00081         otaPrintln("  Max. HW Ver:    0x%2X", header.maximumHardwareVersion);
00082         emberAfOtaBootloadClusterFlush();
00083       }
00084       if (EMBER_AF_OTA_STORAGE_SUCCESS == emAfOtaStorageReadAllTagInfo(&id,
00085                                                                        tagInfo,
00086                                                                        MAX_TAGS,
00087                                                                        &totalTags)) {
00088         int16u i;
00089         otaPrintln("  Total Tags: %d", totalTags);
00090         for (i = 0; i < MAX_TAGS && i < totalTags; i++) {
00091           otaPrintln("    Tag: 0x%2X", tagInfo[i].id);
00092           otaPrintln("      Length: %l", tagInfo[i].length);
00093         }
00094         emberAfOtaBootloadClusterFlush();
00095       } else {
00096         otaPrintln("Error: Could not obtain tag info from image.");
00097       }
00098     }
00099     id = emberAfOtaStorageIteratorNextCallback();
00100     i++;
00101   }
00102 
00103   otaPrintln("\n%d images in OTA storage.", emberAfOtaStorageGetCountCallback());
00104 }

00106 EmberAfOtaImageId emAfOtaFindImageIdByIndex(int8u index)
00107 {
00108   int8u i = 0;
00109   EmberAfOtaImageId id = emberAfOtaStorageIteratorFirstCallback();
00110   do {
00111     if (i == index) {
00112       return id;
00113     }
00114     i++;
00115     id = emberAfOtaStorageIteratorNextCallback();
00116   } while (emberAfIsOtaImageIdValid(&id));

00118   return emberAfInvalidImageId;
