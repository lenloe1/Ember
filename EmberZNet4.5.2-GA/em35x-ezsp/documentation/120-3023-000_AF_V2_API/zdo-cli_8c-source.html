00134 // For simple bind requests, just put {} as the last argument
00135 //   zdo bind <dest> <local ep> <remote ep> <cluster> <THEIR EUI> {}
00136 //
00137 // More complex requests, you can actually specify the dest EUI64 of,
00138 // the binding, which is NOT the same as the EUI64 of the destination
00139 // of the device receiving the binding.  This allows for a user
00140 // to set multiple bindings on the destination for different devices.
00141 //   zdo bind <dest> <local ep> <remote ep> <cluster> <THEIR EUI> <dest EUI64>
00142 static void zdoBindCommand(void)
00143 {
00144   EmberStatus status;
00145   EmberEUI64 sourceEui, destEui;  // names relative to binding sent over-the-air
00146   EmberNodeId target = emberUnsignedCommandArgument(0);
00147   int8u sourceEndpoint = emberUnsignedCommandArgument(1);
00148   int8u destinationEndpoint = emberUnsignedCommandArgument(2);
00149   int16u clusterId = emberUnsignedCommandArgument(3);
00150 
00151   // NOTE:  The source/dest EUI is relative to the context.
00152   // In the case of the syntax of the CLI, we take "THEIR EUI64" to mean
00153   // the recipient of the binding request message.  However, when sending
00154   // the bind request that EUI64 used by the ZDO command is
00155   // actually the source for the binding because we are telling the remote
00156   // device (the destination) to create a binding with a source of itself.
00157   // And the destination for that binding will be this local device.
00158   // This is also not to be confused with the (short) destination of the ZDO
00159   // request itself.  
00160   emberAfCopyBigEndianEui64Argument(4, sourceEui);  // their EUI (the destination's)
00161 
00162   // If the last argument is empty, assume an EUI64 of the local device.
00163   // This allows for the simple case.  If an EUI64 is specified, it will
00164   // be used instead of the local EUI.  This is used for setting
00165   // multiple bindings on the same remote device.
00166   if (0 == emberAfCopyBigEndianEui64Argument(5, destEui)) {
00167     emberAfAppPrintln("Using my local EUI64 for dest EUI64 in binding");
00168     emberAfGetEui64(destEui);
00169   }
00170 
00171   status = emberBindRequest(target,          // who gets the bind req
00172                             sourceEui,       // source eui IN the binding
00173                             sourceEndpoint,
00174                             clusterId,       
00175                             UNICAST_BINDING, // binding type
00176                             destEui,         // destination eui IN the binding
00177                             0,               // groupId for new binding
00178                             destinationEndpoint,
00179                             APS_OPTIONS);
00180   emberAfAppPrintln("ZDO bind req %x", status);
00181 }

00183 static void zdoAddClusterCommand(void)
00184 {
00185   int16u *clusters;
00186   int8u *clCount;
00187   if (emberStringCommandArgument(-2, NULL)[0]  == 'i') {
00188     clusters = zdoInClusters;
00189     clCount = &inClCount;
00190   } else {
00191     clusters = zdoOutClusters;
00192     clCount = &outClCount;
00193   }
00194 
00195   if (*clCount < MAX_CLUSTERS_CAN_MATCH) {
00196     clusters[*clCount] = emberUnsignedCommandArgument(0);
00197     (*clCount)++;
00198   } else {
00199     emberAfAppPrintln("cluster limit reached");
00200   }
00201 }

00282 // Copy the list of int16u input and output cluster lists into
00283 // message buffers.
00284 EmberStatus matchDescriptorsRequest(EmberNodeId target,
00285                                     int16u profile,
00286                                     int8u inCount,
00287                                     int8u outCount,
00288                                     int16u *inClusters,
00289                                     int16u *outClusters,
00290                                     EmberApsOption options)
00291 {
00292   int8u i;
00293   int8u output;
00294   EmberMessageBuffer inClusterBuffer = EMBER_NULL_MESSAGE_BUFFER;
00295   EmberMessageBuffer outClusterBuffer = EMBER_NULL_MESSAGE_BUFFER;
00296   EmberStatus status = EMBER_NO_BUFFERS;
00297   for (output = 0; output < 2; output++) {
00298     EmberMessageBuffer* bufferPtr;
00299     int8u count;
00300     int16u* list;
00301     if (output) {
00302       count = outCount;
00303       list = outClusters;
00304       bufferPtr = &outClusterBuffer;
00305     } else {  // input
00306       count = inCount;
00307       list = inClusters;
00308       bufferPtr = &inClusterBuffer;
00309     }
00310     if (count == 0) {
00311       continue;
00312     }
00313 
00314     *bufferPtr = emberAllocateStackBuffer();
00315     if (*bufferPtr == EMBER_NULL_MESSAGE_BUFFER) {
00316       goto cleanup;
00317     }
00318 
00319     for (i = 0; i < count; i++) {
00320       int8u cluster[2];
00321       cluster[0] = LOW_BYTE(list[i]);
00322       cluster[1] = HIGH_BYTE(list[i]);
00323       status = emberAppendToLinkedBuffers(*bufferPtr,
00324                                           cluster,
00325                                           2);
00326       if (EMBER_SUCCESS != status) {
00327         goto cleanup;
00328       }
00329     }
00330   }
00331   status = emberMatchDescriptorsRequest(target,
00332                                         profile,
00333                                         inClusterBuffer,
00334                                         outClusterBuffer,
00335                                         options);
00336  cleanup:
00337   if (inClusterBuffer != EMBER_NULL_MESSAGE_BUFFER) {
00338     emberReleaseMessageBuffer(inClusterBuffer);
00339   }
00340   if (outClusterBuffer != EMBER_NULL_MESSAGE_BUFFER) {
00341     emberReleaseMessageBuffer(outClusterBuffer);
00342   }
00343   return status;
00344 }
