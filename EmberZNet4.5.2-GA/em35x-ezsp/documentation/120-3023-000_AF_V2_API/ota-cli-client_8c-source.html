00132 static void otaCliBootload(void)
00133 {
00134   int8u index = emberUnsignedCommandArgument(0);
00135   EmberAfOtaImageId id = emAfOtaFindImageIdByIndex(index);
00136   if (!emberAfIsOtaImageIdValid(&id)) {
00137     otaPrintln("Error: No image at index %d", index);
00138     return;
00139   }
00140   emberAfOtaClientBootloadCallback(&id);
00141 }
